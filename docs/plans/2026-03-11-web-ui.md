# WorkFort Web UI — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `workfort web` command that serves a SolidJS browser UI for all WorkFort platform services, starting with the Sharkfin chat integration.

**Architecture:** Go CLI gets a new Cobra subcommand (`web`) following hexagonal architecture. The Go server embeds a SolidJS SPA via `go:embed` and reverse-proxies API/WebSocket traffic to backend services (Sharkfin, Nexus, Hive). Services are organized into "Forts" — named gateway + service collections. The frontend speaks the Sharkfin envelope protocol directly through the proxy.

**Tech Stack:** Go 1.25 (net/http, httputil, go:embed), SolidJS, UnoCSS (Tailwind preset), @solidjs/router, nanostores, Vite, pnpm

**Spec:** `docs/2026-03-11-web-ui-design.md`

```go
// internal/domain/web.go
package domain

// Fort is a named collection of services behind a single gateway.
type Fort struct {
	Name     string
	Gateway  string // Single origin URL for remote forts; empty for local.
	Services []Service
}

// IsLocal returns true if this fort has no remote gateway (the CLI is the gateway).
func (f Fort) IsLocal() bool {
	return f.Gateway == ""
}

// Service represents a backend service within a fort.
type Service struct {
	Name     string
	PathBase string   // Derived: "/api/" + Name
	URL      string   // Direct backend URL (local fort only).
	WSPaths  []string // Paths that accept WebSocket upgrade (whitelist).
	Enabled  bool
}

// NewService creates a Service with a derived PathBase.
func NewService(name, url string, wsPaths []string, enabled bool) Service {
	return Service{
		Name:     name,
		PathBase: "/api/" + name,
		URL:      url,
		WSPaths:  wsPaths,
		Enabled:  enabled,
	}
}

// HasWSPath returns true if path is in the WebSocket whitelist.
func (s Service) HasWSPath(path string) bool {
	for _, p := range s.WSPaths {
		if p == path {
			return true
		}
	}
	return false
}

// FortRegistry provides access to configured forts.
type FortRegistry interface {
	Forts() []Fort
	Active() Fort
	SetActive(name string) error
}
```

```go
// internal/app/web_service.go
package app

import "github.com/Work-Fort/WorkFort/internal/domain"

// ServiceInfo is the JSON representation of a service for the frontend.
type ServiceInfo struct {
	Name    string `json:"name"`
	Label   string `json:"label"`
	Route   string `json:"route"`
	Enabled bool   `json:"enabled"`
}

// ServicesResp is the JSON response for GET /api/services.
type ServicesResp struct {
	Fort     string        `json:"fort"`
	Services []ServiceInfo `json:"services"`
}

// ConfigResp is the JSON response for GET /api/config.
type ConfigResp struct {
	Username string `json:"username"`
	Fort     string `json:"fort"`
}

// labelMap maps service names to display labels.
var labelMap = map[string]string{
	"sharkfin": "Chat",
}

// routeMap maps service names to client-side route prefixes.
var routeMap = map[string]string{
	"sharkfin": "/chat",
}

// WebService is the application service for the web UI.
type WebService struct {
	registry domain.FortRegistry
	username string
}

// NewWebService creates a WebService.
func NewWebService(registry domain.FortRegistry, username string) *WebService {
	return &WebService{registry: registry, username: username}
}

// Registry returns the underlying FortRegistry.
func (s *WebService) Registry() domain.FortRegistry {
	return s.registry
}

// ServicesResponse builds the GET /api/services response.
func (s *WebService) ServicesResponse() ServicesResp {
	fort := s.registry.Active()
	services := make([]ServiceInfo, len(fort.Services))
	for i, svc := range fort.Services {
		label := svc.Name
		if l, ok := labelMap[svc.Name]; ok {
			label = l
		}
		route := "/" + svc.Name
		if r, ok := routeMap[svc.Name]; ok {
			route = r
		}
		services[i] = ServiceInfo{
			Name:    svc.Name,
			Label:   label,
			Route:   route,
			Enabled: svc.Enabled,
		}
	}
	return ServicesResp{Fort: fort.Name, Services: services}
}

// ConfigResponse builds the GET /api/config response.
func (s *WebService) ConfigResponse() ConfigResp {
	return ConfigResp{
		Username: s.username,
		Fort:     s.registry.Active().Name,
	}
}
```

```go
// internal/infra/fortconfig/registry.go
package fortconfig

import (
	"fmt"
	"sort"

	"github.com/spf13/viper"

	"github.com/Work-Fort/WorkFort/internal/domain"
)

// Registry implements domain.FortRegistry backed by Viper config.
type Registry struct {
	v *viper.Viper
}

// NewRegistry creates a FortRegistry reading from the given Viper instance.
func NewRegistry(v *viper.Viper) *Registry {
	return &Registry{v: v}
}

// Active returns the currently active fort.
func (r *Registry) Active() domain.Fort {
	name := r.v.GetString("active-fort")
	return r.loadFort(name)
}

// Forts returns all configured forts.
func (r *Registry) Forts() []domain.Fort {
	sub := r.v.Sub("forts")
	if sub == nil {
		return nil
	}
	settings := sub.AllSettings()
	forts := make([]domain.Fort, 0, len(settings))
	for name := range settings {
		forts = append(forts, r.loadFort(name))
	}
	sort.Slice(forts, func(i, j int) bool { return forts[i].Name < forts[j].Name })
	return forts
}

// SetActive changes the active fort name.
func (r *Registry) SetActive(name string) error {
	// Verify the fort exists
	key := "forts." + name
	if !r.v.IsSet(key) {
		return fmt.Errorf("fort %q not found", name)
	}
	r.v.Set("active-fort", name)
	return nil
}

func (r *Registry) loadFort(name string) domain.Fort {
	prefix := "forts." + name
	fort := domain.Fort{
		Name:    name,
		Gateway: r.v.GetString(prefix + ".gateway"),
	}

	svcSub := r.v.Sub(prefix + ".services")
	if svcSub == nil {
		return fort
	}

	settings := svcSub.AllSettings()
	services := make([]domain.Service, 0, len(settings))
	for svcName := range settings {
		svcPrefix := prefix + ".services." + svcName
		svc := domain.NewService(
			svcName,
			r.v.GetString(svcPrefix+".url"),
			r.v.GetStringSlice(svcPrefix+".ws-paths"),
			r.v.GetBool(svcPrefix+".enabled"),
		)
		services = append(services, svc)
	}
	sort.Slice(services, func(i, j int) bool { return services[i].Name < services[j].Name })
	fort.Services = services
	return fort
}
```

```go
	"net/http"
	"net/http/httputil"
	"net/url"
	"strings"

	"github.com/Work-Fort/WorkFort/internal/domain"
)

// NewServiceProxy returns an http.Handler that reverse-proxies requests
// to the given service, stripping the service's PathBase prefix.
func NewServiceProxy(svc domain.Service) http.Handler {
	if !svc.Enabled {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			w.Header().Set("Content-Type", "application/json")
			w.WriteHeader(http.StatusServiceUnavailable)
			json.NewEncoder(w).Encode(map[string]string{
				"error":   "service disabled",
				"service": svc.Name,
			})
		})
	}

	target, _ := url.Parse(svc.URL)

	proxy := &httputil.ReverseProxy{
		Rewrite: func(r *httputil.ProxyRequest) {
			r.SetURL(target)
			// Strip the PathBase prefix: /api/nexus/v1/vms → /v1/vms
			r.Out.URL.Path = strings.TrimPrefix(r.Out.URL.Path, svc.PathBase)
			if r.Out.URL.Path == "" {
				r.Out.URL.Path = "/"
			}
		},
	}

	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		// Check WebSocket upgrade against whitelist
		if isWebSocketUpgrade(r) {
			relPath := strings.TrimPrefix(r.URL.Path, svc.PathBase)
			if !svc.HasWSPath(relPath) {
				http.Error(w, "WebSocket upgrade not allowed on this path", http.StatusBadRequest)
				return
			}
		}
		proxy.ServeHTTP(w, r)
	})
}

func isWebSocketUpgrade(r *http.Request) bool {
	return strings.EqualFold(r.Header.Get("Upgrade"), "websocket")
}
```

```go
// internal/infra/httpapi/handler.go
package httpapi

import (
	"encoding/json"
	"io/fs"
	"net/http"

	"github.com/Work-Fort/WorkFort/internal/app"
)

// NewHandler creates the top-level HTTP handler.
// If devMode is true, static file serving is skipped (Vite handles it).
func NewHandler(svc *app.WebService, devMode bool) http.Handler {
	mux := http.NewServeMux()

	// API endpoints
	mux.HandleFunc("GET /api/services", func(w http.ResponseWriter, r *http.Request) {
		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(svc.ServicesResponse())
	})

	mux.HandleFunc("GET /api/config", func(w http.ResponseWriter, r *http.Request) {
		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(svc.ConfigResponse())
	})

	// Service proxies
	fort := svc.Registry().Active()
	for _, s := range fort.Services {
		proxy := NewServiceProxy(s)
		mux.Handle(s.PathBase+"/", proxy)
	}

	// SPA fallback (skip in dev mode — Vite serves the frontend)
	if !devMode {
		distFS, err := DistFS()
		if err != nil {
			// Fallback: serve a simple error page
			mux.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
				http.Error(w, "frontend not built", http.StatusInternalServerError)
			})
		} else {
			mux.Handle("/", spaHandler(distFS))
		}
	}

	return mux
}

// spaHandler serves static files, falling back to index.html for client-side routing.
func spaHandler(fsys fs.FS) http.Handler {
	fileServer := http.FileServer(http.FS(fsys))
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		// Try to serve the file directly
		path := r.URL.Path
		if path == "/" {
			path = "index.html"
		}
		_, err := fs.Stat(fsys, path[1:]) // strip leading /
		if err != nil {
			// File not found — serve index.html for SPA routing
			r.URL.Path = "/"
		}
		fileServer.ServeHTTP(w, r)
	})
}
```

```go
package web

import (

	"github.com/Work-Fort/WorkFort/internal/infra/fortconfig"
	"github.com/Work-Fort/WorkFort/internal/infra/httpapi"
)

var (
	bind    string
	port    int
	devMode bool
	noOpen  bool
)

func NewWebCmd() *cobra.Command {
	cmd := &cobra.Command{
		Use:   "web",
		Short: "Open the web UI",
		Long:  "Start an HTTP server serving the WorkFort web interface.",
		RunE:  runWeb,
	}

	cmd.Flags().StringVar(&bind, "bind", "127.0.0.1", "Address to bind")
	cmd.Flags().IntVar(&port, "port", 8080, "Port to listen on")
	cmd.Flags().BoolVar(&devMode, "dev", false, "Dev mode (skip embedded file serving)")
	cmd.Flags().BoolVar(&noOpen, "no-open", false, "Don't auto-open browser")

	return cmd
}

func runWeb(cmd *cobra.Command, args []string) error {
	username := viper.GetString("username")
	if username == "" {
		return fmt.Errorf("username is required (--username flag or WORKFORT_USERNAME env)")
	}

	// Composition root: config → domain → app → infra
	registry := fortconfig.NewRegistry(viper.GetViper())
	svc := app.NewWebService(registry, username)
	handler := httpapi.NewHandler(svc, devMode)

	addr := fmt.Sprintf("%s:%d", bind, port)
	server := &http.Server{
		Addr:    addr,
		Handler: handler,
	}

	// Graceful shutdown
	ctx, stop := signal.NotifyContext(context.Background(), os.Interrupt, syscall.SIGTERM)
	defer stop()

	go func() {
		<-ctx.Done()
		shutdownCtx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
		defer cancel()
		log.Info("shutting down web server")
		server.Shutdown(shutdownCtx)
	}()

	url := fmt.Sprintf("http://%s", addr)
	if bind == "127.0.0.1" || bind == "0.0.0.0" {
		url = fmt.Sprintf("http://localhost:%d", port)
	}

	log.Info("starting web server", "addr", addr, "url", url)
	fmt.Fprintf(os.Stderr, "WorkFort web UI: %s\n", url)

	if !noOpen && !devMode {
		openBrowser(url)
	}

	if err := server.ListenAndServe(); err != http.ErrServerClosed {
		return fmt.Errorf("server error: %w", err)
	}

	return nil
}

func openBrowser(url string) {
	var cmd *exec.Cmd
	switch runtime.GOOS {
	case "darwin":
		cmd = exec.Command("open", url)
	case "windows":
		cmd = exec.Command("cmd", "/c", "start", url)
	default:
		cmd = exec.Command("xdg-open", url)
	}
	// Best effort — don't fail if browser can't be opened
	_ = cmd.Start()
}
```

```ts
export class ApiError extends Error {
  constructor(public status: number, message: string) {
    super(message);
  }

async function request<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(path, {
    ...init,
    headers: { Accept: "application/json", ...init?.headers },
  });
  if (!res.ok) {
    throw new ApiError(res.status, `${res.status} ${res.statusText}`);
  }
  return res.json();
}
```

```ts
// web/src/lib/ws.ts

// --- Envelope types (mirrors pkg/sharkfin/messages.go) ---

export interface Envelope {
  type: string;
  d?: unknown;
  ref?: string;
  ok?: boolean;
}

export interface Hello {
  heartbeat_interval: number;
}

export interface Channel {
  name: string;
  public: boolean;
  member: boolean;
}

export interface Message {
  id: number;
  channel: string;
  from: string;
  body: string;
  sent_at: string;
  thread_id?: number;
}

export interface MessageNewEvent {
  id: number;
  channel: string;
  channel_type: string;
  from: string;
  body: string;
  sent_at: string;
  thread_id?: number;
  mentions?: string[];
}

export interface User {
  username: string;
  online: boolean;
}

export interface PresenceEvent {
  username: string;
  online: boolean;
}

export interface ChannelUnreadCount {
  channel: string;
  type: string;
  unread_count: number;
  mention_count: number;
}

export interface DM {
  channel: string;
  participant?: string;
  participants?: string[];
  member: boolean;
}

// --- Client ---

export type MessageHandler = (type: string, data: unknown) => void;

export class SharkfinClient {
  private ws: WebSocket | null = null;
  private refSeq = 0;
  private pendingRefs = new Map<string, string>(); // ref → request type
  private onMessage: MessageHandler;
  private onConnect: () => void;
  private onDisconnect: (err?: Event) => void;
  private reconnectTimer: ReturnType<typeof setTimeout> | null = null;
  private closed = false;

  constructor(opts: {
    onMessage: MessageHandler;
    onConnect: () => void;
    onDisconnect: (err?: Event) => void;
  }) {
    this.onMessage = opts.onMessage;
    this.onConnect = opts.onConnect;
    this.onDisconnect = opts.onDisconnect;
  }

  connect(username: string) {
    this.closed = false;
    const protocol = window.location.protocol === "https:" ? "wss:" : "ws:";
    const ws = new WebSocket(`${protocol}//${window.location.host}/api/sharkfin/ws`);

    ws.onopen = () => {
      // Wait for hello — handled in onmessage
    };

    let handshakePhase: "hello" | "identify" | "register" | "ready" = "hello";

    ws.onmessage = (event) => {
      const env: Envelope = JSON.parse(event.data);

      if (handshakePhase === "hello") {
        if (env.type === "hello") {
          handshakePhase = "identify";
          const ref = this.nextRef();
          ws.send(JSON.stringify({
            type: "identify",
            d: { username, password: "" },
            ref,
          }));
        }
        return;
      }

      if (handshakePhase === "identify") {
        if (env.ok === true) {
          handshakePhase = "ready";
          this.ws = ws;
          this.onConnect();
        } else if (env.ok === false) {
          // Identify failed — try register as a single fallback attempt
          handshakePhase = "register";
          const ref = this.nextRef();
          ws.send(JSON.stringify({
            type: "register",
            d: { username, password: "" },
            ref,
          }));
        }
        return;
      }

      if (handshakePhase === "register") {
        if (env.ok === true) {
          handshakePhase = "ready";
          this.ws = ws;
          this.onConnect();
        } else {
          // Both identify and register failed — close connection
          ws.close();
        }
        return;
      }

      // Ready — dispatch messages
      this.dispatchMessage(env);
    };

    ws.onclose = (event) => {
      if (!this.closed) {
        this.onDisconnect(event);
      }
    };

    ws.onerror = () => {
      // onclose will fire after onerror
    };
  }

  close() {
    this.closed = true;
    if (this.reconnectTimer) clearTimeout(this.reconnectTimer);
    this.ws?.close();
    this.ws = null;
  }

  reconnect(username: string, attempt: number) {
    const delay = Math.min(1000 * Math.pow(2, attempt - 1), 30000);
    this.reconnectTimer = setTimeout(() => {
      this.connect(username);
    }, delay);
  }

  // --- Send methods ---

  requestChannels() { this.send("channel_list", null); }
  requestHistory(channel: string, before: number, limit: number) {
    this.send("history", { channel, before, limit });
  }
  requestUnread(channel: string) { this.send("unread_messages", { channel }); }
  requestUsers() { this.send("user_list", null); }
  sendMessage(channel: string, body: string) { this.send("send_message", { channel, body }); }
  requestUnreadCounts() { this.send("unread_counts", null); }
  requestDMList() { this.send("dm_list", null); }
  dmOpen(username: string) { this.send("dm_open", { username }); }
  markRead(channel: string, messageId?: number) {
    this.send("mark_read", { channel, message_id: messageId });
  }

  // --- Internal ---

  private send(type: string, data: unknown) {
    if (!this.ws || this.ws.readyState !== WebSocket.OPEN) return;
    const ref = this.nextRef();
    this.pendingRefs.set(ref, type);
    this.ws.send(JSON.stringify({ type, d: data, ref }));
  }

  private nextRef(): string {
    return `ref_${++this.refSeq}`;
  }

  private dispatchMessage(env: Envelope) {
    switch (env.type) {
      case "heartbeat":
        this.send("heartbeat", null);
        break;
      case "reply":
        this.dispatchReply(env);
        break;
      case "message.new":
      case "presence":
        this.onMessage(env.type, env.d);
        break;
      default:
        break;
    }
  }

  private dispatchReply(env: Envelope) {
    const reqType = this.pendingRefs.get(env.ref!);
    if (!reqType) return;
    this.pendingRefs.delete(env.ref!);

    if (env.ok === false) return;

    this.onMessage(`reply:${reqType}`, env.d);
  }
}
```

```ts
// web/src/stores/chat.ts
import { atom } from "nanostores";
import { SharkfinClient, type Channel, type Message, type User, type DM, type ChannelUnreadCount, type MessageNewEvent, type PresenceEvent } from "../lib/ws";
import { $connection, $reconnectAttempt } from "./connection";
import { $username } from "./fort";

export const $channels = atom<Channel[]>([]);
export const $dms = atom<DM[]>([]);
export const $users = atom<User[]>([]);
export const $activeChannel = atom<string>("");
export const $messages = atom<Message[]>([]);
export const $unreadCounts = atom<Record<string, { unread: number; mentions: number }>>({});

let client: SharkfinClient | null = null;

export function initChat() {
  const username = $username.get();
  if (!username) return;

  client = new SharkfinClient({
    onConnect: () => {
      $connection.set("connected");
      $reconnectAttempt.set(0);
      // Request initial data
      client!.requestChannels();
      client!.requestUsers();
      client!.requestUnreadCounts();
      client!.requestDMList();
    },
    onDisconnect: () => {
      $connection.set("connecting");
      const attempt = $reconnectAttempt.get() + 1;
      $reconnectAttempt.set(attempt);
      client!.reconnect(username, attempt);
    },
    onMessage: handleMessage,
  });

  $connection.set("connecting");
  client.connect(username);
}

export function destroyChat() {
  client?.close();
  client = null;
  $connection.set("disconnected");
}

export function sendChatMessage(body: string) {
  const channel = $activeChannel.get();
  if (!channel || !body.trim()) return;
  client?.sendMessage(channel, body.trim());
}

export function selectChannel(name: string) {
  $activeChannel.set(name);
  $messages.set([]);
  client?.requestHistory(name, 0, 50);
  client?.markRead(name);
  // Clear unread for this channel
  const counts = { ...$unreadCounts.get() };
  delete counts[name];
  $unreadCounts.set(counts);
}

export function openDM(username: string) {
  client?.dmOpen(username);
}

function handleMessage(type: string, data: unknown) {
  switch (type) {
    case "reply:channel_list": {
      const d = data as { channels: Channel[] };
      $channels.set(d.channels.filter((c) => c.member));
      break;
    }
    case "reply:history": {
      const d = data as { channel: string; messages: Message[] };
      if (d.channel === $activeChannel.get()) {
        $messages.set(d.messages);
      }
      break;
    }
    case "reply:user_list": {
      const d = data as { users: User[] };
      $users.set(d.users);
      break;
    }
    case "reply:unread_counts": {
      const d = data as { counts: ChannelUnreadCount[] };
      const counts: Record<string, { unread: number; mentions: number }> = {};
      for (const c of d.counts) {
        if (c.unread_count > 0 || c.mention_count > 0) {
          counts[c.channel] = { unread: c.unread_count, mentions: c.mention_count };
        }
      }
      $unreadCounts.set(counts);
      break;
    }
    case "reply:dm_list": {
      const d = data as { dms: DM[] };
      $dms.set(d.dms);
      break;
    }
    case "reply:dm_open": {
      const d = data as { channel: string; participant: string };
      selectChannel(d.channel);
      break;
    }
    case "message.new": {
      const msg = data as MessageNewEvent;
      if (msg.channel === $activeChannel.get()) {
        $messages.set([...$messages.get(), {
          id: msg.id,
          channel: msg.channel,
          from: msg.from,
          body: msg.body,
          sent_at: msg.sent_at,
          thread_id: msg.thread_id,
        }]);
        client?.markRead(msg.channel, msg.id);
      } else {
        // Increment unread count
        const counts = { ...$unreadCounts.get() };
        const cur = counts[msg.channel] ?? { unread: 0, mentions: 0 };
        counts[msg.channel] = {
          unread: cur.unread + 1,
          mentions: cur.mentions + (msg.mentions?.includes($username.get()) ? 1 : 0),
        };
        $unreadCounts.set(counts);
      }
      break;
    }
    case "presence": {
      const evt = data as PresenceEvent;
      $users.set(
        $users.get().map((u) =>
          u.username === evt.username ? { ...u, online: evt.online } : u
        )
      );
      break;
    }
  }
}
```

Go through each component and add `light:` counterparts to dark-only utility classes. The pattern:

| Dark class | Light counterpart |
|-----------|-------------------|
| `border-zinc-800` | `light:border-zinc-200` |
| `bg-zinc-800/60` | `light:bg-zinc-100/60` |
| `bg-zinc-900/50` | `light:bg-zinc-50` |
| `text-zinc-400` | `light:text-zinc-600` |
| `text-zinc-500` | `light:text-zinc-500` (same) |
| `text-zinc-600` | `light:text-zinc-400` |
| `bg-zinc-700` | `light:bg-zinc-200` |
| `text-zinc-300` | `light:text-zinc-700` |
